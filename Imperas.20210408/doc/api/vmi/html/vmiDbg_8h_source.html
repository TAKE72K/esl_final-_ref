223 //
224 // Structure used to pass information about the registers in a simulated
225 // processor to the simulator
226 //
227 typedef struct vmiRegInfoS {
228  const char *name; // register identification name
229  const char *description; // description string
230  vmiRegGroupCP group; // group for this register
231  Uns32 gdbIndex; // gdb ordinal index number
232  vmiRegUsage usage : 3; // any special usage for this register
233  vmiRegAccess access : 2; // allowed access
234  Bool noSaveRestore : 1; // does not participate in save/restore
235  Bool noTraceChange : 1; // does not participate in tracing
236  Bool instrAttrIgnore : 1; // ignore for instruction attributes
237  Bool isAlias : 1; // register an alias of another?
238  Bool extension : 1; // defined in an extension library?
239  Uns32 unused : 6; // unused fields
240  Uns32 bits : 16; // size of register in bits
241  vmiRegReadFn readCB; // read callback function
242  vmiRegWriteFn writeCB; // write callback function
243  vmiReg raw; // raw register value (if no callback)
244  void *userData; // model-specific data pointer
245 } vmiRegInfo;
246 
247 //
248 // Special readCB and writeCB values specifying that read and write of raw
249 // register values is allowed (i.e. without callback), requiring that either:
250 // 1. the 'raw' field is set with the appropriate register; or:
251 // 2. the userData argument is set to the register offset in the processor
252 // structure (deprecated backwards-compatible mode).
253 // With the introduction of the 'raw' field in the vmiRegInfo structure, these
254 // special values are not required (specify the 'raw' vmiReg and omit the the
255 // read/write callback to achieve the same effect).
256 //
257 #define VMI_REG_RAW_READ_CB ((vmiRegReadFn)(-1))
258 #define VMI_REG_RAW_WRITE_CB ((vmiRegWriteFn)(-1))
